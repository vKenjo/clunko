🎯 EXAMPLE: Buy a Ticket via Stacks.js

```javascript
import { makeContractCall, broadcastTransaction } from '@stacks/transactions';

const buyTicket = await makeContractCall({
  contractAddress: 'ST1PQHQKV0RJXZFY1DGX8MNSNYVE3VGZJSRTPGZGM',
  contractName: 'main-lottery',
  functionName: 'buy-ticket',
  functionArgs: [
    listCV([uintCV(5), uintCV(12), uintCV(23), uintCV(34), uintCV(42), uintCV(56)]),
    noneCV()
  ],
  network: new StacksTestnet(),
  senderKey: 'your-private-key',
  fee: 10000,
  postConditionMode: PostConditionMode.Deny,
  postConditions: [
    makeStandardSTXPostCondition(
      'your-address',
      FungibleConditionCode.Equal,
      1000000 // 1 STX
    )
  ]
});

const txId = await broadcastTransaction(buyTicket, network);
console.log('Transaction ID:', txId);
```
